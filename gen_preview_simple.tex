def calcular_fibonacci(n):
    """Calcula la sucesi√≥n de Fibonacci"""
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    
    fib = [0, 1]
    while len(fib) < n:
        fib.append(fib[-1] + fib[-2])
    return fib

# Ejemplo de uso
resultado = calcular_fibonacci(10)
print(f"Fibonacci: {resultado}")
